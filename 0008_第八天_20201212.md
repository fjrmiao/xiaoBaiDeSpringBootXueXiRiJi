# 用上created
my.vue
```
<template>
<div>
  <p>My Page</p>
  <h2>{{title}}</h2>
    <div v-on:click="ChangeMessage">{{message}}</div>
</div>
</template>

<script>
export default{
    name:'My',
    data(){
        return{
        x:-1,
        my:['有朋自远方来','好人一生平安'],
        title:'名言',
        message:''
        }
    },
    methods:{
        ChangeMessage:function(){
            this.x=this.x+1;
            this.message=this.my[this.x%2];
        }
    },
    created:function(){
        this.ChangeMessage();
    }
}
</script>

<style scoped>

</style>
```
